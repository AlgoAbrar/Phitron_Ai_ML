    # 👉 dhore nilam sob integer (CF e mostly eta thake)
    nums = list(map(int, data))  # sob value ke integer e convert korbo

    total = sum(nums)  # sob number jog korbo
    maximum = max(nums)  # maximum ber korbo
    minimum = min(nums)  # minimum ber korbo

    # example condition
    if total % 2 == 0:  # jodi total even hoy
        print("YES")  # YES print korbo
    else:
        print("NO")  # NO print korbo
